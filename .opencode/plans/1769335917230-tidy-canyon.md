# Raw Download Streaming API Implementation Plan

## Overview

Add a new REST API endpoint `/api/v1/agents/{agent}/raw/{*path}` that streams file content in chunks via WebSocket binary messages instead of loading the entire file into memory like the existing `cat` endpoint.

## Binary Protocol Design

### Chunk Format

Each binary message contains a 24-byte header followed by the data payload:

```
[Header: 24 bytes][Data: N bytes]

Header (little-endian):
- magic:          4 bytes (0x52415844 = "RAXD")
- request_id:     8 bytes (u64)
- chunk_index:    8 bytes (u64)
- is_last:        1 byte (0x01 = last chunk, 0x00 = more chunks)
- flags:          1 byte (reserved, must be 0)
- reserved:       2 bytes (must be 0)
```

async fn raw_download(&self, path: String) -> CommandResult {
    match tokio::fs::File::open(&path).await {
        Ok(mut file) => {
            let mut chunk_index = 0u64;
            let chunk_size = CHUNK_SIZE;
            let mut buffer = vec![0u8; chunk_size];

            loop {
                match file.read(&mut buffer).await {
                    Ok(0) => break,
                    Ok(n) => {
                        // Send chunk via WebSocket
                        // (request_id injected by caller)
                        chunk_index += 1;
                    }
                    Err(e) => {
                        return CommandResult::Error {
                            message: format!("Failed to read file: {}", e),
                        };
                    }
                }
            }
            CommandResult::RawDownload { path }
        }
        Err(e) => CommandResult::Error {
            message: format!("Failed to open file: {}", e),
        },
    }
}

RouterMsg::RouteStreamChunk {
    agent_id,
    request_id,
    chunk_index,
    is_last,
    data,
} => {
    if let Some(chunk_sender) = state.rest_streaming_responses.remove(&request_id) {
        let chunk = crate::streaming::StreamChunk {
            request_id,
            chunk_index,
            is_last,
            data,
        };

        if !is_last {
            state.rest_streaming_responses.insert(request_id, chunk_sender.clone());
        }

        let _ = chunk_sender.send(chunk.to_bytes());

        if is_last {
            log!(
                Level::Info,
                "Streaming complete: agent_id={}, request_id={}, total_chunks={}",
                agent_id,
                request_id,
                chunk_index + 1
            );
        }
    } else {
        log!(
            Level::Warning,
            "No streaming response found for request_id={}",
            request_id
        );
    }
}

RouterMsg::ExecuteStreamCommandRest {
    agent_id,
    command,
    reply,
    chunk_sender,
} => {
    let request_id = state.next_id();

    log!(
        Level::Info,
        "Routing REST streaming command: agent_id={}, request_id={}, command={:?}",
        agent_id,
        request_id,
        command
    );

    if let Some(agent_info) = state.agents.get(&agent_id) {
        state.rest_streaming_responses.insert(request_id, chunk_sender);

        let _ = agent_info.session_ref.cast(SessionMsg::OutgoingMessage(
            Message::Command {
                agent_id: agent_id.clone(),
                request_id,
                command,
            },
        ));

        let _ = reply.send(());
    } else {
        let _ = reply.send(());
    }
}

async fn raw_download(&self, path: String, request_id: u64, write: &mpsc::UnboundedSender<WsMessage>) -> CommandResult {
    match tokio::fs::File::open(&path).await {
        Ok(mut file) => {
            let mut chunk_index = 0u64;
            let chunk_size = crate::streaming::CHUNK_SIZE;
            let mut buffer = vec![0u8; chunk_size];

            loop {
                match file.read(&mut buffer).await {
                    Ok(0) => break,
                    Ok(n) => {
                        let chunk = crate::streaming::StreamChunk {
                            request_id,
                            chunk_index,
                            is_last: false,
                            data: buffer[..n].to_vec(),
                        };
                        let _ = write.send(WsMessage::Binary(chunk.to_bytes().into()));
                        chunk_index += 1;
                    }
                    Err(e) => {
                        return CommandResult::Error {
                            message: format!("Failed to read file: {}", e),
                        };
                    }
                }
            }

            // Send final chunk
            let final_chunk = crate::streaming::StreamChunk {
                request_id,
                chunk_index,
                is_last: true,
                data: Vec::new(),
            };
            let _ = write.send(WsMessage::Binary(final_chunk.to_bytes().into()));

            CommandResult::RawDownload { path }
        }
        Err(e) => CommandResult::Error {
            message: format!("Failed to open file: {}", e),
        },
    }
}

```rust
async fn raw_agent_handler(
    Path((agent, path)): Path<(String, String)>,
    AxumState(state): AxumState<ServerState>,
) -> impl IntoResponse {
    let (response_sender, mut response_receiver) = tokio::sync::mpsc::unbounded_channel::<Vec<u8>>();

    match call_t!(
        &state.router_ref,
        |reply| actors::router::RouterMsg::ExecuteStreamCommandRest {
            agent_id: agent.clone(),
            command: Command::RawDownload { path: path.clone() },
            reply,
            chunk_sender: response_sender,
        },
        30000
    ) {
        Ok(()) => {
            let body = axum::body::Body::from_stream(async_stream::stream! {
                while let Some(chunk) = response_receiver.recv().await {
                    // Strip header from last chunk and send only data
                    if chunk.len() >= crate::streaming::HEADER_SIZE {
                        let is_last = chunk[20] == 1;
                        if is_last {
                            // Send only the data portion (skip header)
                            if chunk.len() > crate::streaming::HEADER_SIZE {
                                yield Ok::<_, axum::Error>(bytes::Bytes::from(&chunk[crate::streaming::HEADER_SIZE..]));
                            }
                        } else {
                            // Send only the data portion (skip header)
                            yield Ok::<_, axum::Error>(bytes::Bytes::from(&chunk[crate::streaming::HEADER_SIZE..]));
                        }
                    }
                }
            });

            Response::builder()
                .status(StatusCode::OK)
                .header("Content-Type", "application/octet-stream")
                .header("Content-Disposition", format!("attachment; filename=\"{}\"",
                    path.split('/').last().unwrap_or("file")))
                .body(body)
                .unwrap()
                .into_response()
        }
        Err(e) => {
            let error_msg = format!("Failed to start stream: {:?}", e);
            (
                StatusCode::INTERNAL_SERVER_ERROR,
                Json(ErrorResponse { error: error_msg }),
            ).into_response()
        }
    }
}
```

async raw(path: string): Promise<ArrayBuffer> {
    const url = `${this.baseUrl}/api/v1/agents/${encodeURIComponent(this.info.id)}/raw/${encodeURIComponent(path)}`;
    const response = await fetch(url);

    if (!response.ok) {

            throw new Error(error.error);

import { spawn, ChildProcess } from 'node:child_process'
import { describe, it, expect, beforeAll, afterAll } from 'vitest'

const WS_URL = `ws://127.0.0.1:${SERVER_PORT}/ws`

class ProcessManager {
  private processes: Map<number, ChildProcess> = new Map()

  spawn(command: string, args: string[], cwd?: string): number {
    const proc = spawn(command, args, {
      detached: true,
      stdio: ['ignore', 'pipe', 'pipe'],
      cwd,
    })

    const pid = proc.pid
    if (pid === undefined) {
      throw new Error('Failed to get process PID')
    }

    this.processes.set(pid, proc)
    return pid
  }

  kill(pid: number): void {
    try {
      process.kill(pid, 'SIGKILL')
    } catch (e) {
      if ((e as NodeJS.ErrnoException).code !== 'ESRCH') {
        throw e
      }
    }
    this.processes.delete(pid)
  }

  killAll(): void {
    for (const pid of this.processes.keys()) {
      this.kill(pid)
    }
  }

  getProcess(pid: number): ChildProcess | undefined {
    return this.processes.get(pid)
  }
}

async function waitForPort(port: number, maxRetries = 50): Promise<void> {
  for (let i = 0; i < maxRetries; i++) {
    try {
      const response = await fetch(`http://127.0.0.1:${port}/`)
      if (response.ok) {
        return
      }
    } catch {
      await new Promise(resolve => setTimeout(resolve, 100))
    }
  }
  throw new Error(`Port ${port} not ready after ${maxRetries} retries`)
}

async function waitForLogMessage(
  process: ChildProcess,
  pattern: RegExp,
  timeoutMs: number = 10000
): Promise<void> {
  const stream = process.stdout || process.stderr
  if (!stream) {
    throw new Error('No stdout/stderr stream available')
  }

  let resolve: () => void
  let reject: (error: Error) => void

  const promise = new Promise<void>((res, rej) => {
    resolve = res
    reject = rej
  })

  const onData = (chunk: Buffer) => {
    const line = chunk.toString()
    if (pattern.test(line)) {
      clearTimeout(timeout)
      stream.off('data', onData)
      resolve()
    }
  }

  stream.on('data', onData)

  const timeout = setTimeout(() => {
    stream.off('data', onData)
    reject(new Error(`Timeout waiting for log pattern: ${pattern}`))
  }, timeoutMs)

  return promise
}

## Error Handling Considerations

1. File not found: Agent returns `CommandResult::Error`, REST handler returns 404
2. Permission denied: Agent returns error, REST handler returns 403
3. Connection lost during stream: RouterActor cleans up pending streams
4. Agent disconnects: RouterActor removes from rest_streaming_responses
5. Timeout: 30-second timeout for starting stream (no timeout during streaming)
